\DAsection[任务分支、小步提交、同步 main、合并检查]{GitHub 小团队协作流程}

\begin{frame}{3.1 小团队开发的完整流程}{一个任务、一个分支、一个合并目标}
  \centering
  \begin{tikzpicture}[node distance=2mm]
    \node[DAflow,minimum width=18mm] (task) {任务\\明确边界};
    \node[DAflow,minimum width=18mm,right=of task] (branch) {Branch\\隔离修改};
    \node[DAflow,minimum width=18mm,right=of branch] (commit) {Commit\\记录变化};
    \node[DAflow,minimum width=18mm,right=of commit] (push) {Push\\共享分支};
    \node[DAflow,minimum width=18mm,right=of push] (review) {PR\\简单 Review};
    \node[DAflow,minimum width=18mm,right=of review] (merge) {Merge\\进入 main};
    \draw[DAarrowred] (task) -- (branch);
    \draw[DAarrowred] (branch) -- (commit);
    \draw[DAarrowred] (commit) -- (push);
    \draw[DAarrowred] (push) -- (review);
    \draw[DAarrowred] (review) -- (merge);
  \end{tikzpicture}
  \par\medskip
  \begin{itemize}
    \item 任务先明确负责人、修改范围与完成条件。
    \item 每个任务使用独立分支，Commit 保持小而清晰。
    \item 推送到在线仓库后，由同伴快速检查并合并到 \DAfile{main}。
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{3.2 开发者的日常命令流}{同步 main、创建分支、提交、推送}
  \begin{lstlisting}[language=bash]
git switch main
git pull --ff-only
git switch -c feature/chat-history

# edit and test
git add src/
git diff --cached
git commit -m "feat(chat): load conversation history"

git push -u origin feature/chat-history
  \end{lstlisting}
  \DAtagline{下一步}{在 GitHub 发起合并请求，由一名同伴快速检查后合并}
\end{frame}

\begin{frame}{3.3 合并前快速检查}{围绕本次任务检查逻辑、测试、配置与密钥}
  \begin{DAtable}{@{}p{29mm}p{88mm}@{}}
    \toprule
    检查项 & 小团队做法 \\
    \midrule
    改动范围 & 确认只包含当前任务，排除无关格式化与临时文件 \\
    核心逻辑 & 检查错误处理、接口变化和关键业务路径 \\
    测试结果 & 运行构建、测试或最小可行验证，记录检查结果 \\
    配置安全 & 排除 \DAfile{.env}、密码、密钥和本地专用配置 \\
    合并收尾 & 检查通过后进入 \DAfile{main}，删除已完成的功能分支 \\
    \bottomrule
  \end{DAtable}
  \DAtagline{控制范围}{一次合并只解决一个问题，降低检查与回滚成本}
\end{frame}

\begin{frame}[fragile]{3.4 开发期间同步 main}{小团队优先使用 Merge，规则保持统一}
  \begin{lstlisting}[language=bash]
git fetch origin
git switch feature/chat-history
git merge origin/main

# resolve conflicts, test, commit
git push
  \end{lstlisting}
  \begin{itemize}
    \item 功能分支开发超过半天时主动同步 \DAfile{main}。
    \item 冲突在自己的分支解决并测试，避免把问题留到最终合并。
    \item 团队熟悉历史重写后，再统一讨论是否使用 Rebase。
  \end{itemize}
\end{frame}

\begin{frame}{3.5 密钥必须留在 Git 之外}{泄漏后立即轮换并清理历史}
  \begin{alertblock}{严禁提交}
    \DAfile{.env}、数据库密码、JWT Secret、云厂商密钥、模型 API Key、私钥文件。
  \end{alertblock}
  \begin{itemize}
    \item 发现泄漏后立即吊销或轮换密钥，再处理仓库文件。
    \item 清理 Git 历史、Fork、缓存和 CI 日志，并检查泄漏原因。
    \item 部署时通过服务器环境变量、Secret 文件或平台 Secret 注入。
  \end{itemize}
  \DAtagline{提交前检查}{\DAcmd{git diff --cached} + Secret Scanner + 最小权限密钥}
\end{frame}
