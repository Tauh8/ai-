\DAsection[任务分支、小步提交、同步 main、合并检查]{GitHub 小团队协作流程}

\begin{frame}{3.1 小团队开发的完整流程}{一个任务、一个分支、一个合并目标}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \centering
      \DAframed{\includegraphics[height=47mm]{assets/official/collected/git-topic-branches.png}}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{enumerate}
        \item 明确任务边界与负责人。
        \item 从 \DAfile{main} 创建短生命周期分支。
        \item 小步 Commit，并 Push 共享。
        \item 同伴快速检查后合并。
        \item 删除已完成的功能分支。
      \end{enumerate}
    \end{column}
  \end{columns}
  \DAsource{Pro Git · Topic Branches}{https://git-scm.com/book/en/v2/Git-Branching-Branching-Workflows}
\end{frame}

\begin{frame}[fragile]{3.2 开发者的日常命令流}{同步 main、创建分支、提交、推送}
  \begin{lstlisting}[language=bash]
git switch main
git pull --ff-only
git switch -c feature/chat-history

# edit and test
git add src/
git diff --cached
git commit -m "feat(chat): load conversation history"

git push -u origin feature/chat-history
  \end{lstlisting}
  \DAtagline{下一步}{在 GitHub 发起合并请求，由一名同伴快速检查后合并}
\end{frame}

\begin{frame}{3.3 合并前快速检查}{围绕本次任务检查逻辑、测试、配置与密钥}
  \centering
  \DAframed{\includegraphics[width=58mm]{assets/official/collected/git-basic-merge.png}}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item 改动只包含当前任务。
        \item 核心逻辑与错误处理正确。
        \item 构建、测试或最小验证通过。
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{itemize}
        \item 不包含 \DAfile{.env}、密码和密钥。
        \item 接口、配置和迁移变化已说明。
        \item 合并后删除功能分支。
      \end{itemize}
    \end{column}
  \end{columns}
  \DAtagline{控制范围}{一次合并只解决一个问题，降低检查与回滚成本}
  \DAsource{Pro Git · Basic Merging}{https://git-scm.com/book/en/v2/Git-Branching-Basic-Branching-and-Merging}
\end{frame}

\begin{frame}[fragile]{3.4 开发期间同步 main}{小团队优先使用 Merge，规则保持统一}
  \centering
  \includegraphics[width=104mm]{assets/official/collected/git-long-running-branches.png}
  \begin{lstlisting}[language=bash]
git fetch origin
git switch feature/chat-history
git merge origin/main

# resolve conflicts, test, commit
git push
  \end{lstlisting}
  \begin{itemize}
    \item 功能分支开发超过半天时主动同步 \DAfile{main}。
    \item 冲突在自己的分支解决并测试，避免把问题留到最终合并。
  \end{itemize}
  \DAsource{Pro Git · Long-Running Branches}{https://git-scm.com/book/en/v2/Git-Branching-Branching-Workflows}
\end{frame}

\begin{frame}{3.5 密钥必须留在 Git 之外}{泄漏后立即轮换并清理历史}
  \begin{alertblock}{严禁提交}
    \DAfile{.env}、数据库密码、JWT Secret、云厂商密钥、模型 API Key、私钥文件。
  \end{alertblock}
  \begin{itemize}
    \item 发现泄漏后立即吊销或轮换密钥，再处理仓库文件。
    \item 清理 Git 历史、Fork、缓存和 CI 日志，并检查泄漏原因。
    \item 部署时通过服务器环境变量、Secret 文件或平台 Secret 注入。
  \end{itemize}
  \DAtagline{提交前检查}{\DAcmd{git diff --cached} + Secret Scanner + 最小权限密钥}
\end{frame}
