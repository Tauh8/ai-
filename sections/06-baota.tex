\begin{frame}{6.1 环境组件与职责}{先理解每一层负责什么}
  \begin{DAtable}{@{}p{28mm}p{38mm}p{55mm}@{}}
    \toprule
    组件 & 用途 & 选择建议 \\
    \midrule
    Nginx & 80 / 443、静态文件、反向代理 & 必装 \\
    Node.js & 构建前端、运行 Node 后端 / SSR & 安装项目要求的 LTS 版本 \\
    pnpm & Node 依赖与构建 & 通过 Corepack 或官方方式启用 \\
    Python + uv & FastAPI 运行环境 & 与 \DAfile{pyproject.toml} 要求一致 \\
    MySQL & 业务数据 & 本机部署时仅监听本地 / 内网 \\
    \bottomrule
  \end{DAtable}
  \vspace{1mm}
  {\small\textcolor{DARed}{\bfseries 原则：}项目版本来自仓库声明，生产升级要有计划。}
\end{frame}

\begin{frame}{6.2 静态前端在宝塔里的本质}{Nginx 读取构建产物，不直接运行前端源码}
  \begin{itemize}
    \item 网站根目录指向 \DAfile{dist/} 等生产构建产物，而非 \DAfile{src/} 或项目根目录。
    \item Vite / React / Vue 的静态产物由 Nginx 直接返回，不需要常驻 Node 进程。
    \item 前端请求使用相对路径 \DAfile{/api}，方便由同域 Nginx 转发后端。
    \item 构建时环境变量会写入浏览器可见的静态文件，不能保存 Secret。
    \item 环境变量或源码变化后，需要重新构建并替换产物。
  \end{itemize}
  \DAtagline{先判断}{静态构建与 Next.js / Node SSR 的运行方式不同}
\end{frame}

\begin{frame}[fragile]{6.3 SPA 刷新 404 怎么修}{让未知前端路由回到 index.html}
  \begin{lstlisting}[language=bash]
location / {
    try_files $uri $uri/ /index.html;
}
  \end{lstlisting}
  \begin{itemize}
    \item React Router / Vue Router 的 History 模式由浏览器管理路由。
    \item 用户直接访问 \DAfile{/chat/123} 时，Nginx 找不到同名文件会返回 404。
    \item \DAcmd{try\_files} 让前端路由接管；\DAfile{/api} 保持独立代理规则。
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{6.6 Nginx 把 /api 转给 FastAPI}{同域部署减少 CORS 与 Cookie 问题}
  \begin{lstlisting}[language=bash]
location /api/ {
    proxy_pass http://127.0.0.1:8000/;
    proxy_set_header Host $host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
}
  \end{lstlisting}
  \begin{itemize}
    \item 访问 \DAfile{https://app.example.com/api/health}，Nginx 转给 FastAPI。
    \item 注意 \DAcmd{proxy\_pass} 末尾斜杠会影响路径改写，必须现场验证真实路由。
    \item 后端启用 Proxy Headers 后，才能正确感知外部协议与客户端信息。
  \end{itemize}
  \DAsource{NGINX Docs · Reverse Proxy}{https://docs.nginx.com/nginx/admin-guide/web-server/reverse-proxy/}
\end{frame}

\begin{frame}{6.7 反向代理界面要看懂的字段}{面板字段最终都会变成 Nginx 配置}
  \begin{columns}[T]
    \begin{column}{0.42\textwidth}
      \begin{itemize}
        \item 代理目录决定哪些请求被匹配。
        \item 目标 URL 指向本机应用端口。
        \item Host 与转发头保留域名、协议和客户端信息。
        \item API 通常关闭缓存；静态资源可单独配置缓存。
        \item WebSocket / SSE 需要额外确认连接与超时设置。
      \end{itemize}
    \end{column}

\begin{frame}{6.8 前后端联调的验证顺序}{后端、Nginx、浏览器逐层检查}
  \begin{enumerate}
    \item \DAcmd{curl http://127.0.0.1:8000/health}：后端进程是否正常。
    \item \DAcmd{curl -i http://127.0.0.1/api/health}：Nginx 反代是否正常。
    \item 浏览器打开域名：前端静态资源是否正常。
    \item F12 Network 查看 \DAfile{/api/...} 请求、状态码、响应体。
    \item 后端日志确认同一请求是否到达、耗时与错误。
  \end{enumerate}
  \DAtagline{定位法}{在哪一层第一次失败，问题就位于该层或它的下一跳}
\end{frame}

\begin{frame}{6.9 Node 后端与 FastAPI 的部署差异}{运行时不同，部署模型相同}
  \begin{DAtable*}{@{}p{28mm}p{42mm}p{45mm}@{}}
    \toprule
    对比项 & Node 后端 & FastAPI 后端 \\
    \midrule
    依赖锁定 & \DAfile{pnpm-lock.yaml} & \DAfile{uv.lock} \\
    构建阶段 & NestJS / TypeScript 常需构建 & 通常直接运行 Python 源码 \\
    运行时 & Node.js + pnpm / PM2 & Python 虚拟环境 + ASGI Server \\
    本机端口 & 常见 \DAfile{127.0.0.1:3000} & 常见 \DAfile{127.0.0.1:8000} \\
    进程管理 & 宝塔 Node 项目 / PM2 & Python 项目 / 应用管理器 \\
    公网入口 & \multicolumn{2}{l}{都由 Nginx 反向代理，不直接暴露业务端口} \\
    \bottomrule
  \end{DAtable*}
  \DAtagline{共同模型}{锁定依赖 + 本机端口 + 进程守护 + Nginx 入口}
\end{frame}

\begin{frame}{6.11 环境变量如何放到服务器}{仓库保存变量名，服务器保存真实值}
  \begin{DAtable}{@{}p{35mm}p{82mm}@{}}
    \toprule
    类型 & 处理方式 \\
    \midrule
    前端公开配置 & 构建时注入；默认视为用户可见，不放 Secret \\
    后端普通配置 & 项目环境变量或权限受控的 Env 文件 \\
    数据库密码 / API Key & 最小权限、定期轮换、日志脱敏 \\
    JWT Secret & 使用足够随机值，变更会影响现有 Token \\
    上传目录 & 独立持久目录，权限给运行用户并纳入备份 \\
    \bottomrule
  \end{DAtable}
  \DAtagline{权限}{Env 文件只允许部署用户 / 运行用户读取，避免通过静态站点暴露}
\end{frame}
